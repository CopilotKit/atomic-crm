# Interactive Guided Demo Mode — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a driver.js-powered interactive guided demo that walks through the four CopilotKit workflows (account review, contract risk scan, renewal forecast, audit log) via a state machine activated by `?demo=guided`.

**Architecture:** A `useReducer`-based state machine (S0–S6) drives the demo flow. driver.js renders spotlight/popover narration for each state. `DemoProvider` lives inside `Layout.tsx` (inside Router + CopilotKit context). Query params `?demo=guided` and `?autoAgent=true` control activation and auto-triggering.

**Tech Stack:** React, driver.js ^1.3.1, CopilotKit v2 (useAgent, useCopilotKit), React Router (useNavigate, useSearchParams), Vitest

// src/components/atomic-crm/demo/useDemoStateMachine.ts
import { useReducer, useEffect, useCallback, useRef, useState } from "react";
import type { DemoState } from "./demoSteps";

export type DemoStateOrDone = DemoState | "DONE";
export type AgentPhase = "idle" | "running" | "done";

export interface DemoMachineState {
  state: DemoStateOrDone;
  agentPhase: AgentPhase;
  errorCount: number;
}

export type DemoAction =
  | { type: "ADVANCE" }
  | { type: "RESET" }
  | { type: "AGENT_STARTED" }
  | { type: "AGENT_FINISHED" }
  | { type: "AGENT_ERROR" }
  | { type: "SKIP_STATE" };

const STATE_ORDER: DemoStateOrDone[] = [
  "S0_IDLE",
  "S1_OPEN_CONTACT",
  "S2_AGENT_REVIEW",
  "S3_CONTRACT_ANALYSIS",
  "S4_FORECAST_PROPOSAL",
  "S5_APPROVAL_PENDING",
  "S6_AUDIT_LOG",
  "DONE",
];

function nextState(current: DemoStateOrDone): DemoStateOrDone {
  const idx = STATE_ORDER.indexOf(current);
  if (idx === -1 || idx >= STATE_ORDER.length - 1) return "DONE";
  return STATE_ORDER[idx + 1];
}

export const initialState: DemoMachineState = {
  state: "S0_IDLE",
  agentPhase: "idle",
  errorCount: 0,
};

export function demoReducer(
  state: DemoMachineState,
  action: DemoAction,
): DemoMachineState {
  switch (action.type) {
    case "ADVANCE":
      return {
        state: nextState(state.state),
        agentPhase: "idle",
        errorCount: 0,
      };
    case "RESET":
      return initialState;
    case "AGENT_STARTED":
      return { ...state, agentPhase: "running" };
    case "AGENT_FINISHED":
      return { ...state, agentPhase: "done" };
    case "AGENT_ERROR":
      return { ...state, errorCount: state.errorCount + 1, agentPhase: "idle" };
    case "SKIP_STATE":
      return {
        state: nextState(state.state),
        agentPhase: "idle",
        errorCount: 0,
      };
    default:
      return state;
  }
}

/** Whether the user can click "Next" for the given state. */
export function computeCanAdvance(
  machineState: DemoMachineState,
  elementReady?: boolean,
): boolean {
  switch (machineState.state) {
    case "S0_IDLE":
      return true;
    case "S1_OPEN_CONTACT":
      return elementReady === true;
    case "S2_AGENT_REVIEW":
    case "S3_CONTRACT_ANALYSIS":
      return machineState.agentPhase === "done";
    case "S4_FORECAST_PROPOSAL":
      return false; // auto-transitions, no manual advance
    case "S5_APPROVAL_PENDING":
      // S5 becomes advanceable when the HITL card disappears (approve/reject
      // triggers agent to process the response, which flips isRunning true→false).
      // We track this via agentPhase just like S2/S3.
      return machineState.agentPhase === "done";
    case "S6_AUDIT_LOG":
      return true;
    case "DONE":
      return false;
    default:
      return false;
  }
}

const AGENT_STATES: DemoStateOrDone[] = [
  "S2_AGENT_REVIEW",
  "S3_CONTRACT_ANALYSIS",
  "S4_FORECAST_PROPOSAL",
];

// States where agentPhase tracking applies (S2/S3/S4 for agent runs, S5 for HITL response)
const AGENT_PHASE_STATES: DemoStateOrDone[] = [
  ...AGENT_STATES,
  "S5_APPROVAL_PENDING",
];

interface UseDemoStateMachineOptions {
  autoAgent: boolean;
  triggerAgent: (prompt: string) => Promise<void>;
  navigate: (path: string) => void;
  agentIsRunning: boolean;
  contactRoute: string; // e.g. "/contacts/123/show"
  prompts: Record<string, string>; // keyed by state name
}

export function useDemoStateMachine({
  autoAgent,
  triggerAgent,
  navigate,
  agentIsRunning,
  contactRoute,
  prompts,
}: UseDemoStateMachineOptions) {
  const [machineState, dispatch] = useReducer(demoReducer, initialState);
  const prevAgentRunning = useRef(false);
  const [elementReady, setElementReady] = useState(false);

  // Track agent running → finished transitions
  // Applies to S2/S3/S4 (agent workflows) and S5 (HITL approve/reject triggers agent)
  useEffect(() => {
    if (!AGENT_PHASE_STATES.includes(machineState.state)) {
      prevAgentRunning.current = agentIsRunning;
      return;
    }
    if (agentIsRunning && machineState.agentPhase === "idle") {
      dispatch({ type: "AGENT_STARTED" });
    }
    if (!agentIsRunning && prevAgentRunning.current && machineState.agentPhase === "running") {
      dispatch({ type: "AGENT_FINISHED" });
    }
    prevAgentRunning.current = agentIsRunning;
  }, [agentIsRunning, machineState.state, machineState.agentPhase]);

  // On state entry: navigate and/or trigger agent
  const prevState = useRef(machineState.state);
  useEffect(() => {
    if (prevState.current === machineState.state) return;
    prevState.current = machineState.state;
    setElementReady(false); // reset for new state

    // Navigate if needed
    if (machineState.state === "S1_OPEN_CONTACT") {
      navigate(contactRoute);
    } else if (machineState.state === "S6_AUDIT_LOG") {
      navigate("/audit");
    } else if (machineState.state === "DONE") {
      navigate("/");
    }

    // Auto-trigger agent if enabled
    if (autoAgent && AGENT_STATES.includes(machineState.state)) {
      const prompt = prompts[machineState.state];
      if (prompt) {
        triggerAgent(prompt);
      }
    }
  }, [machineState.state, autoAgent, contactRoute, navigate, prompts, triggerAgent]);

  const advance = useCallback(() => dispatch({ type: "ADVANCE" }), []);
  const reset = useCallback(() => {
    dispatch({ type: "RESET" });
    navigate("/");
  }, [navigate]);
  const skipState = useCallback(() => dispatch({ type: "SKIP_STATE" }), []);
  const reportError = useCallback(() => dispatch({ type: "AGENT_ERROR" }), []);

  return {
    state: machineState.state,
    agentPhase: machineState.agentPhase,
    errorCount: machineState.errorCount,
    canAdvance: computeCanAdvance(machineState, elementReady),
    elementReady,
    setElementReady,
    advance,
    reset,
    skipState,
    reportError,
    dispatch,
  };
}
```

// src/components/atomic-crm/demo/useDemoDriver.ts
import { useEffect, useRef, useCallback } from "react";
import { driver as createDriver, type Driver } from "driver.js";
import "driver.js/dist/driver.css";
import type { DemoStateOrDone, AgentPhase } from "./useDemoStateMachine";
import { getDemoSteps, type DemoStepDef } from "./demoSteps";
import { DOM_POLL_TIMEOUT_MS, DOM_POLL_INTERVAL_MS, MAX_AGENT_RETRIES } from "./demoConfig";

/**
 * Poll for a DOM element matching the selector.
 * Resolves with the element or null after timeout.
 */
export function waitForElement(
  selector: string,
  timeoutMs: number = DOM_POLL_TIMEOUT_MS,
): Promise<Element | null> {
  return new Promise((resolve) => {
    const existing = document.querySelector(selector);
    if (existing) {
      resolve(existing);
      return;
    }

    const start = Date.now();
    const interval = setInterval(() => {
      const el = document.querySelector(selector);
      if (el) {
        clearInterval(interval);
        resolve(el);
      } else if (Date.now() - start >= timeoutMs) {
        clearInterval(interval);
        resolve(null);
      }
    }, DOM_POLL_INTERVAL_MS);
  });
}

interface UseDemoDriverOptions {
  state: DemoStateOrDone;
  agentPhase: AgentPhase;
  canAdvance: boolean;
  errorCount: number;
  autoAgent: boolean;
  advance: () => void;
  reset: () => void;
  skipState: () => void;
  reportError: () => void;
  setElementReady: (ready: boolean) => void;
}

export function useDemoDriver({
  state,
  agentPhase,
  canAdvance,
  errorCount,
  autoAgent,
  advance,
  reset,
  skipState,
  reportError,
  setElementReady,
}: UseDemoDriverOptions) {
  const driverRef = useRef<Driver | null>(null);
  const stateRef = useRef(state);
  stateRef.current = state;

  // Keep refs to callbacks so driver popover buttons always call the latest version
  const advanceRef = useRef(advance);
  advanceRef.current = advance;
  const resetRef = useRef(reset);
  resetRef.current = reset;
  const skipRef = useRef(skipState);
  skipRef.current = skipState;
  const retryRef = useRef(reportError);
  retryRef.current = reportError;

  const canAdvanceRef = useRef(canAdvance);
  canAdvanceRef.current = canAdvance;
  const errorCountRef = useRef(errorCount);
  errorCountRef.current = errorCount;

  // Create driver instance once
  useEffect(() => {
    driverRef.current = createDriver({
      allowClose: false,
      overlayClickNext: false,
      allowKeyboardControl: false,
      stagePadding: 8,
      stageRadius: 8,
    });
    return () => {
      driverRef.current?.destroy();
      driverRef.current = null;
    };
  }, []);

  const steps = getDemoSteps(autoAgent);
  // Also get the autoAgent steps for re-targeting after agent completes
  const autoAgentSteps = getDemoSteps(true);

  // Highlight the current state's element
  const highlightState = useCallback(
    async (demoState: DemoStateOrDone, stepOverride?: DemoStepDef) => {
      const d = driverRef.current;
      if (!d || demoState === "DONE") {
        driverRef.current?.destroy();
        return;
      }

      const stepDef = stepOverride ?? steps[demoState as keyof typeof steps];
      if (!stepDef) return;

      if (stepDef.element === null) {
        // S0: centered modal via drive() — no element target
        d.drive({
          steps: [{
            popover: {
              title: stepDef.title,
              description: stepDef.description,
              onPopoverRender: (popover: any) => {
                renderPopoverButtons(popover.wrapper, stepDef, errorCountRef);
              },
            },
          }],
        });
        setElementReady(true);
        return;
      }

      // Wait for element to appear in DOM
      const el = await waitForElement(stepDef.element);
      if (!el) {
        // Element not found — skip this state
        skipRef.current();
        return;
      }

      setElementReady(true);

      d.highlight({
        element: stepDef.element,
        popover: {
          title: stepDef.title,
          description: stepDef.description,
          side: stepDef.side,
          onPopoverRender: (popover: any) => {
            renderPopoverButtons(popover.wrapper, stepDef, errorCountRef);
          },
        },
      });
    },
    [steps, setElementReady],
  );

  // React to state changes
  useEffect(() => {
    highlightState(state);
  }, [state, highlightState]);

  // S4 auto-transition: poll for HITL card appearing in DOM, then advance to S5
  useEffect(() => {
    if (state !== "S4_FORECAST_PROPOSAL") return;
    let cancelled = false;

    waitForElement('[data-demo="hitl-card"]').then((el) => {
      if (!cancelled && el) {
        advanceRef.current(); // S4 → S5
      }
    });

    return () => { cancelled = true; };
  }, [state]);

  // Re-target popover when agent finishes in non-autoAgent mode (S2/S3)
  // Switch from button highlight to copilot panel with descriptive text
  useEffect(() => {
    if (autoAgent) return;
    if (agentPhase !== "done") return;
    const agentStates = ["S2_AGENT_REVIEW", "S3_CONTRACT_ANALYSIS"];
    if (!agentStates.includes(state as string)) return;

    // Re-highlight with the autoAgent step definition (points to copilot-panel)
    const autoStep = autoAgentSteps[state as keyof typeof autoAgentSteps];
    if (autoStep) {
      highlightState(state, autoStep);
    }
  }, [state, agentPhase, autoAgent, autoAgentSteps, highlightState]);

  // Update button disabled states and retry/skip when canAdvance or errorCount changes
  useEffect(() => {
    const wrapper = document.querySelector(".driver-popover");
    if (!wrapper) return;

    // Update Next button
    const nextBtn = wrapper.querySelector<HTMLButtonElement>("[data-demo-btn='next']");
    if (nextBtn) {
      nextBtn.disabled = !canAdvance;
      nextBtn.style.opacity = canAdvance ? "1" : "0.5";
    }

    // Update Retry/Skip buttons based on errorCount
    const retryBtn = wrapper.querySelector<HTMLButtonElement>("[data-demo-btn='retry']");
    const skipBtn = wrapper.querySelector<HTMLButtonElement>("[data-demo-btn='skip']");
    if (errorCount > 0 && errorCount < MAX_AGENT_RETRIES) {
      // Show Retry, hide Skip
      if (!retryBtn) {
        const btn = createActionButton("Retry", "retry", () => retryRef.current());
        const btnContainer = wrapper.querySelector("[data-demo-btns]");
        if (btnContainer) btnContainer.prepend(btn);
      }
      if (skipBtn) skipBtn.remove();
    } else if (errorCount >= MAX_AGENT_RETRIES) {
      // Show Skip, hide Retry
      if (retryBtn) retryBtn.remove();
      if (!skipBtn) {
        const btn = createActionButton("Skip", "skip", () => skipRef.current());
        const btnContainer = wrapper.querySelector("[data-demo-btns]");
        if (btnContainer) btnContainer.prepend(btn);
      }
    } else {
      // No errors — remove both
      retryBtn?.remove();
      skipBtn?.remove();
    }
  }, [canAdvance, errorCount]);

  return { driverRef };
}

function createActionButton(label: string, id: string, onClick: () => void): HTMLButtonElement {
  const btn = document.createElement("button");
  btn.textContent = label;
  btn.setAttribute("data-demo-btn", id);
  btn.style.cssText =
    "padding:6px 12px;border-radius:6px;font-size:13px;cursor:pointer;border:1px solid #d1d5db;background:#fff;color:#374151;";
  btn.addEventListener("click", onClick);
  return btn;
}

function renderPopoverButtons(
  wrapper: Element,
  stepDef: DemoStepDef,
  errorCountRef: { current: number },
) {
  // Remove existing driver.js buttons
  const footer = wrapper.querySelector(".driver-popover-footer");
  if (footer) footer.remove();

  const btnContainer = document.createElement("div");
  btnContainer.setAttribute("data-demo-btns", "");
  btnContainer.style.cssText =
    "display:flex;gap:8px;justify-content:flex-end;margin-top:12px;padding-top:12px;border-top:1px solid #e5e7eb;";

  // Exit Demo button
  const exitBtn = document.createElement("button");
  exitBtn.textContent = "Exit Demo";
  exitBtn.style.cssText =
    "padding:6px 12px;border-radius:6px;font-size:13px;cursor:pointer;border:1px solid #d1d5db;background:#fff;color:#374151;";
  exitBtn.addEventListener("click", () => {
    const resetFn = (window as any).__demoReset;
    if (resetFn) resetFn();
  });
  btnContainer.appendChild(exitBtn);

  // Next / Finish button (hidden for auto-transition states)
  if (!stepDef.autoTransition) {
    const nextBtn = document.createElement("button");
    nextBtn.textContent = stepDef.advanceLabel ?? "Next";
    nextBtn.setAttribute("data-demo-btn", "next");
    nextBtn.style.cssText =
      "padding:6px 12px;border-radius:6px;font-size:13px;cursor:pointer;border:none;background:#2563eb;color:#fff;opacity:0.5;";
    nextBtn.disabled = true;
    nextBtn.addEventListener("click", () => {
      const advanceFn = (window as any).__demoAdvance;
      if (advanceFn) advanceFn();
    });
    btnContainer.appendChild(nextBtn);
  }

  wrapper.appendChild(btnContainer);
}
```

Note: The hook exposes button callbacks via `window.__demoAdvance` and `window.__demoReset` because driver.js popover buttons are rendered outside React's tree. These are set/cleaned up by `DemoProvider`.

// src/components/atomic-crm/demo/DemoProvider.tsx
import { useEffect, useMemo, useCallback, useState, type ReactNode } from "react";
import { useSearchParams, useNavigate, useLocation } from "react-router";
import { useAgent, useCopilotKit } from "@copilotkit/react-core/v2";
import { randomUUID } from "@copilotkit/shared";
import { useDataProvider } from "ra-core";
import { DemoContext, type DemoContextValue } from "./DemoContext";
import { useDemoStateMachine } from "./useDemoStateMachine";
import { useDemoDriver } from "./useDemoDriver";
import { DEMO_CONTACT, DEMO_PROMPTS } from "./demoConfig";

export function DemoProvider({ children }: { children: ReactNode }) {
  const [searchParams, setSearchParams] = useSearchParams();
  const isDemoMode = searchParams.get("demo") === "guided";
  const autoAgent = isDemoMode && searchParams.get("autoAgent") === "true";

  if (!isDemoMode) {
    return <DemoInactiveProvider>{children}</DemoInactiveProvider>;
  }

  return (
    <DemoActiveProvider autoAgent={autoAgent} setSearchParams={setSearchParams}>
      {children}
    </DemoActiveProvider>
  );
}

/** No-op context when demo is off — zero overhead */
function DemoInactiveProvider({ children }: { children: ReactNode }) {
  const value = useMemo<DemoContextValue>(
    () => ({
      isDemoMode: false,
      autoAgent: false,
      state: "S0_IDLE",
      agentPhase: "idle",
      canAdvance: false,
      errorCount: 0,
      advance: () => {},
      reset: () => {},
      skipState: () => {},
      reportError: () => {},
      requestCopilotTab: false,
    }),
    [],
  );
  return <DemoContext.Provider value={value}>{children}</DemoContext.Provider>;
}

interface DemoActiveProviderProps {
  autoAgent: boolean;
  setSearchParams: ReturnType<typeof useSearchParams>[1];
  children: ReactNode;
}

function DemoActiveProvider({
  autoAgent,
  setSearchParams,
  children,
}: DemoActiveProviderProps) {
  const navigate = useNavigate();
  const location = useLocation();
  const dataProvider = useDataProvider();
  const { agent } = useAgent();
  const { copilotkit } = useCopilotKit();

  // Resolve contact ID
  const [contactId, setContactId] = useState<number | null>(null);
  const [initError, setInitError] = useState<string | null>(null);

  useEffect(() => {
    dataProvider
      .getList("contacts", {
        filter: {
          first_name: DEMO_CONTACT.firstName,
          last_name: DEMO_CONTACT.lastName,
        },
        pagination: { page: 1, perPage: 1 },
        sort: { field: "id", order: "ASC" },
      })
      .then(({ data }) => {
        if (data.length > 0) {
          setContactId(data[0].id as number);
        } else {
          setInitError("Demo contact not found. Import the demo dataset first.");
        }
      })
      .catch(() => {
        setInitError("Failed to resolve demo contact.");
      });
  }, [dataProvider]);

  const contactRoute = contactId ? `/contacts/${contactId}/show` : "";

  const triggerAgent = useCallback(
    async (prompt: string) => {
      agent.addMessage({ id: randomUUID(), role: "user", content: prompt });
      await copilotkit.runAgent({ agent });
    },
    [agent, copilotkit],
  );

  const machine = useDemoStateMachine({
    autoAgent,
    triggerAgent,
    navigate,
    agentIsRunning: agent.isRunning,
    contactRoute,
    prompts: DEMO_PROMPTS,
  });

  useDemoDriver({
    state: machine.state,
    agentPhase: machine.agentPhase,
    canAdvance: machine.canAdvance,
    errorCount: machine.errorCount,
    autoAgent,
    advance: machine.advance,
    reset: machine.reset,
    skipState: machine.skipState,
    reportError: machine.reportError,
    setElementReady: machine.setElementReady,
  });

  // Expose callbacks to driver.js popover buttons (outside React tree)
  useEffect(() => {
    (window as any).__demoAdvance = machine.advance;
    (window as any).__demoReset = () => {
      machine.reset();
      setSearchParams((prev) => {
        const next = new URLSearchParams(prev);
        next.delete("demo");
        next.delete("autoAgent");
        return next;
      });
    };
    return () => {
      delete (window as any).__demoAdvance;
      delete (window as any).__demoReset;
    };
  }, [machine.advance, machine.reset, setSearchParams]);

  // Navigate to / on init (runs once when contactId is resolved)
  const hasInitNavigated = useRef(false);
  useEffect(() => {
    if (contactId && !hasInitNavigated.current && location.pathname !== "/") {
      hasInitNavigated.current = true;
      navigate("/");
    }
  }, [contactId]); // eslint-disable-line react-hooks/exhaustive-deps

  // requestCopilotTab: true when we are about to trigger agent (S2/S3/S4 entry)
  const requestCopilotTab =
    machine.state === "S2_AGENT_REVIEW" ||
    machine.state === "S3_CONTRACT_ANALYSIS" ||
    machine.state === "S4_FORECAST_PROPOSAL" ||
    machine.state === "S5_APPROVAL_PENDING";

  const value = useMemo<DemoContextValue>(
    () => ({
      isDemoMode: true,
      autoAgent,
      state: machine.state,
      agentPhase: machine.agentPhase,
      canAdvance: machine.canAdvance,
      errorCount: machine.errorCount,
      advance: machine.advance,
      reset: machine.reset,
      skipState: machine.skipState,
      reportError: machine.reportError,
      requestCopilotTab,
    }),
    [
      autoAgent,
      machine.state,
      machine.agentPhase,
      machine.canAdvance,
      machine.errorCount,
      machine.advance,
      machine.reset,
      machine.skipState,
      machine.reportError,
      requestCopilotTab,
    ],
  );

  // Show error if contact not found
  if (initError) {
    return (
      <DemoContext.Provider value={value}>
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/50">
          <div className="bg-white rounded-lg p-6 shadow-lg max-w-md">
            <h3 className="font-semibold text-lg mb-2">Demo Error</h3>
            <p className="text-sm text-muted-foreground mb-4">{initError}</p>
            <button
              className="px-4 py-2 bg-blue-600 text-white rounded-md text-sm"
              onClick={() => {
                setSearchParams((prev) => {
                  const next = new URLSearchParams(prev);
                  next.delete("demo");
                  next.delete("autoAgent");
                  return next;
                });
                navigate("/");
              }}
            >
              Close
            </button>
          </div>
        </div>
        {children}
      </DemoContext.Provider>
    );
  }

  return <DemoContext.Provider value={value}>{children}</DemoContext.Provider>;
}
